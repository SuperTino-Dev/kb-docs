## Preinstall step
```




sudo dnf install -y oracle-database-preinstall-19c

Crear directorios

sudo mkdir -p /u01/app/oracle/product/23ai/dbhome_1
sudo mkdir -p /u01/app/oraInventory
sudo mkdir -p /u01/stage
sudo chown -R oracle:oinstall /u01
sudo chmod -R 775 /u01

Cambiar a usuario oracle

su - oracle

Configurar variables de entorno

cat >> ~/.bash_profile <<EOF
export ORACLE_BASE=/u01/app/oracle
export ORACLE_HOME=$ORACLE_BASE/product/23ai/dbhome_1
export ORACLE_SID=JOREL
export PATH=$ORACLE_HOME/bin:$PATH
EOF

source ~/.bash_profile

Descomprimir instalador

cd $ORACLE_HOME 
unzip /u01/stage/V1054592-01.zip


Ejecutar instalación en modo silent

./runInstaller -silent -ignorePrereqFailure \
oracle.install.option=INSTALL_DB_SWONLY \
ORACLE_HOSTNAME=localhost \
UNIX_GROUP_NAME=oinstall \
INVENTORY_LOCATION=/u01/app/oraInventory \
SELECTED_LANGUAGES=en \
oracle.install.db.InstallEdition=EE \
oracle.install.db.OSDBA_GROUP=dba \
oracle.install.db.OSOPER_GROUP=dba \
oracle.install.db.OSBACKUPDBA_GROUP=dba \
oracle.install.db.OSDGDBA_GROUP=dba \
oracle.install.db.OSKMDBA_GROUP=dba \
oracle.install.db.OSRACDBA_GROUP=dba \
DECLINE_SECURITY_UPDATES=true


Ejecutar scripts root (cuando termine)

exit

sudo /u01/app/oraInventory/orainstRoot.sh
sudo /u01/app/oracle/product/23ai/dbhome_1/root.sh

Volver a oracle

su - oracle

mkdir /u01/app/oracle/oradata

Crear base de datos (DBCA silent)

dbca -silent -createDatabase -templateName General_Purpose.dbc -gdbname JOREL -sid JOREL -responseFile NO_VALUE -characterSet AL32UTF8 -sysPassword Krypto28#$ -systemPassword Krypto28#$ -createAsContainerDatabase true -numberOfPDBs 1 -pdbName PDB1 -pdbAdminPassword   Krypto28#$ -databaseType OLTP -memoryMgmtType auto_sga -totalMemory 2048 -storageType FS -datafileDestination "/u01/app/oracle/oradata" -redoLogFileSize 50 -emConfiguration NONE

Levantar listener

lsnrctl start

Entrar a la base

sqlplus / as sysdba

Verificar

select name from v$database;
select name from v$pdbs;

```
